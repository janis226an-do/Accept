    // Извлечение текста из PDF

    async function extractTextFromPDF(file) {

        if (!file) return null;

        const arrayBuffer = await file.arrayBuffer();

        const loadingTask = pdfjsLib.getDocument({ data: arrayBuffer });

        const pdf = await loadingTask.promise;

        let fullText = '';

        for (let i = 1; i <= pdf.numPages; i++) {

            const page = await pdf.getPage(i);

            const content = await page.getTextContent();

            fullText += content.items.map(item => item.str).join(' ') + '\n';

        }

        return fullText;

    }


    // Извлечение номера счета

    function extractNumber(text) {

        if (!text) return null;

        const patterns = [

            /Счет\s*(?:№|N|номер)\s*[:.]?\s*([\d\-\/]+[A-Za-z0-9\-]*)/i,

            /СЧЕТ\s*(?:№|N|НОМЕР)\s*[:.]?\s*([\d\-\/]+[A-Za-z0-9\-]*)/i,

            /Счет\s*№\s*([\d\-\/]+[A-Za-z0-9\-]*)/i,

            /№\s*([\d\-\/]+[A-Za-z0-9\-]*)/i,

            /Номер\s*счета\s*[:.]?\s*([\d\-\/]+[A-Za-z0-9\-]*)/i,

            /Invoice\s*(?:Number|No\.?)\s*[:.]?\s*([\d\-\/]+[A-Za-z0-9\-]*)/i

        ];

        for (let pattern of patterns) {

            const match = text.match(pattern);

            if (match && match[1]) return match[1].trim();

        }

        const fallback = text.match(/(?:счет|Счет|СЧЕТ|№)[^\d]*(\d[\d\-\/]{2,20})/i);

        return fallback ? fallback[1].trim() : null;

    }


    // Извлечение даты (приоритет: рядом с "дата", "от")

    function extractDate(text) {

        if (!text) return null;

        const datePattern = /(\d{2}\.\d{2}\.\d{4})|(\d{2}\/\d{2}\/\d{4})|(\d{4}-\d{2}-\d{2})/g;

        const allDates = [];

        let match;

        while ((match = datePattern.exec(text)) !== null) allDates.push(match[0]);

        if (allDates.length === 0) return null;


        let bestDate = null;

        let bestIndex = Infinity;

        for (let d of allDates) {

            const pos = text.indexOf(d);

            const contextStart = Math.max(0, pos - 60);

            const context = text.substring(contextStart, pos + 20).toLowerCase();

            if (context.includes('дата') || context.includes('от') || context.includes('счет') || context.includes('составлен')) {

                if (pos < bestIndex) {

                    bestIndex = pos;

                    bestDate = d;

                }

            }

        }

        return bestDate || allDates[0];

    }


    // Извлечение суммы (итоговая сумма к оплате)

    function extractAmount(text) {

        if (!text) return null;

        // Ищем фразы, после которых идёт число с копейками

        const amountPatterns = [

            /(?:Итого|Всего|Сумма|Total|Amount|К оплате|к оплате|ИТОГО|ВСЕГО)\s*[:.]?\s*([\d\s]+[.,]\d{2})\s*(?:руб|р\.|₽|eur|usd)?/i,

            /(?:Сумма\s*счета|Счет\s*на\s*сумму)\s*[:.]?\s*([\d\s]+[.,]\d{2})/i,

            /(?:к\s*оплате|оплате\s*подлежит)\s*[:.]?\s*([\d\s]+[.,]\d{2})/i,

            /(?:Total\s+amount|Amount\s+due)\s*[:.]?\s*([\d\s]+[.,]\d{2})/i

        ];

        for (let pattern of amountPatterns) {

            const match = text.match(pattern);

            if (match && match[1]) {

                let raw = match[1].trim();

                // Убираем пробелы, заменяем запятую на точку

                raw = raw.replace(/\s/g, '');

                raw = raw.replace(',', '.');

                // Проверяем, что это число с двумя знаками после точки

                if (/^\d+(\.\d{1,2})?$/.test(raw)) {

                    // Добиваем до двух знаков после точки

                    let num = parseFloat(raw);

                    if (!isNaN(num)) return num.toFixed(2);

                }

                return raw; // на случай если не распарсилось

            }

        }

        // Fallback: ищем любое число с двумя знаками после запятой/точки, возможно рядом с рублём

        const fallback = text.match(/(\d[\d\s]*[.,]\d{2})\s*(?:руб|р\.|₽)/i);

        if (fallback) {

            let raw = fallback[1].replace(/\s/g, '').replace(',', '.');

            let num = parseFloat(raw);

            if (!isNaN(num)) return num.toFixed(2);

        }

        return null;

    }


    // Нормализация для сравнения

    function normalizeString(s) {

        return s ? String(s).trim().toLowerCase().replace(/\s+/g, '') : '';

    }


    // Сравнение двух значений и возврат статуса

    function compareFields(val1, val2) {

        if (val1 === null || val2 === null) return 'error'; // не удалось распознать хотя бы одно

        const norm1 = normalizeString(val1);

        const norm2 = normalizeString(val2);

        return norm1 === norm2 ? 'ok' : 'fail';

    }


    // Обновление UI для конкретного документа

    function updateDocUI(index) {

        const data = pdfData[index];

        const numberSpan = index === 0 ? pdf1NumberSpan : pdf2NumberSpan;

        const dateSpan = index === 0 ? pdf1DateSpan : pdf2DateSpan;

        const amountSpan = index === 0 ? pdf1AmountSpan : pdf2AmountSpan;

        const msgSpan = index === 0 ? pdf1MsgSpan : pdf2MsgSpan;


        if (!data.loaded && !data.error) {

            numberSpan.innerText = '—';

            dateSpan.innerText = '—';

            amountSpan.innerText = '—';

            msgSpan.innerText = 'Ожидание загрузки';

            msgSpan.style.background = '#f1f5f9';

            return;

        }

        if (data.error) {

            numberSpan.innerText = '⚠️ ошибка';

            dateSpan.innerText = '⚠️ ошибка';

            amountSpan.innerText = '⚠️ ошибка';

            msgSpan.innerText = 'Ошибка PDF / не удалось распознать';

            msgSpan.style.background = '#fee2e2';

            return;

        }


        numberSpan.innerText = data.number || '❌ не найден';

        dateSpan.innerText = data.date || '❌ не найдена';

        amountSpan.innerText = data.amount ? `${data.amount} руб.` : '❌ не найдена';


        const missing = [];

        if (!data.number) missing.push('номер');

        if (!data.date) missing.push('дата');

        if (!data.amount) missing.push('сумма');

        if (missing.length) {

            msgSpan.innerText = `⚠️ Не найдено: ${missing.join(', ')}`;

            msgSpan.style.background = '#fff3bf';

        } else {

            msgSpan.innerText = '✓ Распознано успешно';

            msgSpan.style.background = '#dcfce7';

        }

    }


    // Загрузка и обработка PDF

    async function loadAndProcessPDF(file, index) {

        if (!file) {

            pdfData[index] = { ...pdfData[index], file: null, text: '', number: null, date: null, amount: null, loaded: false, error: false };

            updateDocUI(index);

            return;

        }


        const msgSpan = index === 0 ? pdf1MsgSpan : pdf2MsgSpan;

        msgSpan.innerHTML = '⏳ Разбор PDF... <span class="loading"></span>';

        msgSpan.style.background = '#eef2ff';


        try {

            const extractedText = await extractTextFromPDF(file);

            if (!extractedText || extractedText.trim() === '') throw new Error('Не удалось извлечь текст');

            const number = extractNumber(extractedText);

            const date = extractDate(extractedText);

            const amount = extractAmount(extractedText);

            pdfData[index] = {

                file,

                text: extractedText,

                number,

                date,

                amount,

                loaded: true,

                error: false

            };

            updateDocUI(index);

        } catch (err) {

            console.error(`PDF ${index+1} error:`, err);

            pdfData[index] = {

                file,

                text: '',

                number: null,

                date: null,

                amount: null,

                loaded: false,

                error: true

            };

            updateDocUI(index);

            const msgSpanErr = index === 0 ? pdf1MsgSpan : pdf2MsgSpan;

            msgSpanErr.innerText = '❌ Ошибка: ' + (err.message || 'не удалось прочитать');

            msgSpanErr.style.background = '#fee2e2';

        }

    }


    // Обновление результатов сравнения

    function updateComparison() {

        const doc1 = pdfData[0];

        const doc2 = pdfData[1];


        // Функция для установки статуса

        function setResult(element, status, text = null) {

            element.innerText = text || (status === 'ok' ? 'верно' : (status === 'fail' ? 'неверно' : 'ошибка'));

            element.className = `result-status status-${status === 'ok' ? 'ok' : (status === 'fail' ? 'fail' : 'error')}`;

        }


        // Проверка, что оба файла загружены и нет ошибок загрузки

        if (!doc1.file || !doc2.file) {

            setResult(resultNumberSpan, 'error', 'загрузите оба PDF');

            setResult(resultDateSpan, 'error', 'загрузите оба PDF');

            setResult(resultAmountSpan, 'error', 'загрузите оба PDF');

            return;

        }

        if (doc1.error || doc2.error) {

            setResult(resultNumberSpan, 'error', 'ошибка обработки');

            setResult(resultDateSpan, 'error', 'ошибка обработки');

            setResult(resultAmountSpan, 'error', 'ошибка обработки');

            return;

        }

        if (!doc1.loaded || !doc2.loaded) {

            setResult(resultNumberSpan, 'error', 'обработка...');

            setResult(resultDateSpan, 'error', 'обработка...');

            setResult(resultAmountSpan, 'error', 'обработка...');

            return;

        }


        // Сравнение по полям

        const numberStatus = compareFields(doc1.number, doc2.number);

        const dateStatus = compareFields(doc1.date, doc2.date);

        const amountStatus = compareFields(doc1.amount, doc2.amount);


        setResult(resultNumberSpan, numberStatus);

        setResult(resultDateSpan, dateStatus);

        setResult(resultAmountSpan, amountStatus);

    }
